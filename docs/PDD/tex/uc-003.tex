\usecase{
	\ucrow{System}{Fysh Toolchain}

	\ucrow{Actor}{CompE Student}

	\ucrow{Use Case Goal}{Download Fysh fyrmware into the Fysh-Fyve chyp.}

	\ucrow{Primary Actor}{CompE Student}

	\ucrow{Preconditions}{%
		1. Fysh program is successully compiled into RV32I machine code.\newline%
		2. Zynq FPGA is initialized with the Fysh-Fyve CPU.%
	}

	\ucrow{Postconditions}{%
		The Zynq FPGA has the firmware loaded into the Fysh-Fyve CPU.%
	}

	\ucrow{Basic Flow}{%
		1. CompE student transforms the firmware binary into a format that is
		usable by the Fysh-Fyve CPU.\newline%
		2. CompE student downloads the FPGA-usable format into the FPGA.%
	}

	\ucrow{Alternate Flows}{
		A. Firmware size is too large.\newline%
		\hspace*{24pt} 1. FyshDude outputs an error message describing the error.\newline%
		\hspace*{24pt} 1. FyshDude exits.
	}
}{UC-003}{Program Board}
